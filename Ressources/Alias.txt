# Pour utiliser plus facilement l'interpréteur Python en version 3.7

# Pour effacer un fichier de manière sécurisée tant que l'option "-f" n'est pas utilisée
alias rm='rm -i --preserve-root'

# Qualques alias de la commande "ls" très pratiques

alias la='ls -A'
alias ll='ls -alF'
alias lla='ls -l -a'
alias lll='ls -larth'
